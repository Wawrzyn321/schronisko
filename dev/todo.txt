function useVoluteeringFormState() {
  const [state, setState] = useState({
    fullName: '',
    email: '',
    phoneNumber: '',
    birthDate: '',
    about: ''
  })

  const setProperty = (property: keyof typeof state, value: string) => setState(state => ({ ...state, [property]: value }))

  return [state, setProperty]
}
